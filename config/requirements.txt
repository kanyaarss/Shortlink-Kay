# Shortlink Kay v1 - System Requirements
# File ini mendokumentasikan semua dependencies yang diperlukan

# ============================================
# OPERATING SYSTEM
# ============================================
# Ubuntu LTS 24.04.3 (x86_64) atau lebih baru
# Kernel: 6.8.0 atau lebih baru

# ============================================
# WEB SERVER
# ============================================
# Nginx 1.24.0 atau lebih baru
# - HTTP/2 support
# - SSL/TLS support
# - FastCGI caching support

# ============================================
# PHP
# ============================================
# PHP 8.3 atau lebih baru (PHP 8.2 juga supported)

# Required PHP Extensions:
# - php8.3-cli (command line interface)
# - php8.3-fpm (FastCGI Process Manager)
# - php8.3-mysql (MySQL/MariaDB driver)
# - php8.3-pdo (PHP Data Objects)
# - php8.3-json (JSON support)
# - php8.3-curl (cURL support)
# - php8.3-gd (Image processing)
# - php8.3-mbstring (Multibyte string)
# - php8.3-xml (XML support)
# - php8.3-zip (ZIP support)

# Optional PHP Extensions:
# - php8.3-redis (Redis caching)
# - php8.3-memcached (Memcached caching)
# - php8.3-imagick (Advanced image processing)

# ============================================
# DATABASE
# ============================================
# MySQL 8.0+ atau MariaDB 10.11.10+
# - InnoDB engine
# - UTF-8mb4 charset support
# - Prepared statements support

# ============================================
# TOOLS & UTILITIES
# ============================================
# Git 2.34+ (untuk deployment)
# Composer (optional, jika menggunakan dependencies)
# curl (untuk testing API)
# wget (untuk downloading files)

# ============================================
# SYSTEM LIBRARIES (Ubuntu 24.04)
# ============================================
# build-essential
# libssl-dev
# libcurl4-openssl-dev
# libjpeg-dev
# libpng-dev
# libfreetype6-dev
# zlib1g-dev

# ============================================
# INSTALLATION COMMANDS (Ubuntu 24.04)
# ============================================

# Update system
# sudo apt update && sudo apt upgrade -y

# Install Nginx
# sudo apt install nginx -y

# Install PHP 8.3 dan extensions
# sudo apt install php8.3 php8.3-cli php8.3-fpm php8.3-mysql php8.3-pdo php8.3-json php8.3-curl php8.3-gd php8.3-mbstring php8.3-xml php8.3-zip -y

# Install MySQL/MariaDB
# sudo apt install mariadb-server -y

# Install Git
# sudo apt install git -y

# Install utilities
# sudo apt install curl wget build-essential -y

# ============================================
# VERIFICATION COMMANDS
# ============================================

# Check PHP version
# php -v

# Check PHP extensions
# php -m | grep -E 'pdo|mysql|json|curl'

# Check Nginx version
# nginx -v

# Check MySQL version
# mysql --version

# Check Git version
# git --version
